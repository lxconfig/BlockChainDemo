宏的使用：类似于函数 1. 无参数的宏 {% macro input() %} {% endmacro %} {{ input() }}
2. 有参数的宏 {% macro input2(type, value, size) %} {% endmacro %} {{ input2("radio", "", 40) }}
3. 有默认值的宏 {% macro input3(type='password', value='password', size=80) %} {% endmacro %} {{ input3() }}
4. 从文件中引入宏 {% import 'macro_input.html' as func %} {{ func.input3() }}
